CREATE OR REPLACE PROCEDURE P_GET_PROJECTS_BY_TEAMSIZE 
(
  MINSIZE IN NUMBER 
, MAXSIZE IN NUMBER 
, DETAILS OUT SYS_REFCURSOR 
) AS 
BEGIN
  
  OPEN DETAILS FOR
       SELECT PROJID,PROJNAME,DOMAIN,COMPANY,TEAMSIZE  FROM PROJECT  WHERE TEAMSIZE>=MINSIZE AND TEAMSIZE<=MAXSIZE; 
  
END P_GET_PROJECTS_BY_TEAMSIZE;
