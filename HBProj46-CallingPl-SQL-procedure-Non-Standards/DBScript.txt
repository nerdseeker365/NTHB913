create or replace PROCEDURE P_AUTHENTICATION 
(
  UNAME IN VARCHAR2 
, PWD IN VARCHAR2 
, RESULT OUT VARCHAR2 
) AS 
  CNT number(4);
BEGIN
   SELECT COUNT(*) INTO CNT FROM USERINFO WHERE USERNAME=UNAME AND PASSWORD=PWD;
  
  IF(CNT<>0) THEN
      RESULT:='VALID CREDENTIALS';
  ELSE
    RESULT:='INVALID CREDENTIALS';
    END IF;
  
END P_AUTHENTICATION;
